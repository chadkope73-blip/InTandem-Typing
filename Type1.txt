    function randomItem(arr) {
      return arr[Math.floor(Math.random() * arr.length)];
    }

    function normalizeText(text) {
      return text.replace(/\s+/g, ' ').trim();
    }

    function buildPassage(genreKey) {
      const genre = GENRE_BANK[genreKey];
      return normalizeText(`${randomItem(genre.openings)}, ${randomItem(genre.middles)}, ${randomItem(genre.endings)}`);
    }

    function selectedCountry() {
      return COUNTRY_OPTIONS.find(country => country.code === state.countryCode) || COUNTRY_OPTIONS[0];
    }

    function calculateStats() {
      const target = state.passage;
      const typed = state.typed;
      let correct = 0;
      for (let i = 0; i < typed.length; i++) {
        if (typed[i] === target[i]) correct += 1;
      }
      const accuracy = typed.length ? Math.max(0, Math.round((correct / typed.length) * 100)) : 100;
      const wpm = state.elapsed > 0 ? Math.round(((typed.length / 5) / state.elapsed) * 60) : 0;
      const progress = target.length ? Math.min(100, Math.round((typed.length / target.length) * 100)) : 0;
      return { accuracy, wpm, progress };
    }

    function renderFlags() {
      els.flagWrap.innerHTML = '';
      COUNTRY_OPTIONS.forEach(country => {
        const button = document.createElement('button');
        button.type = 'button';
        button.className = `flag-btn ${country.featured ? 'featured' : ''} ${country.code === state.countryCode ? 'active' : ''}`;
        button.textContent = country.flag;
        button.title = `${country.name} — ${country.language}`;
        button.setAttribute('aria-label', `${country.name} ${country.language}`);
        button.addEventListener('click', () => {
          state.countryCode = country.code;
          renderFlags();
          renderSelectedCountry();
          updateStatus();
        });
        els.flagWrap.appendChild(button);
      });
    }

    function renderSelectedCountry() {
      const country = selectedCountry();
      els.selectedCountry.innerHTML = `
        <span class="big-flag">${country.flag}</span>
        <span class="country-name">${country.name}</span>
        <span>•</span>
        <span>${country.language}</span>
      `;
    }

    function renderGenres() {
      els.genreWrap.innerHTML = '';
      Object.entries(GENRE_BANK).forEach(([key, genre]) => {
        const button = document.createElement('button');
        button.type = 'button';
        button.className = `genre-btn ${key === state.genreKey ? 'active' : ''}`;
        button.textContent = genre.label;
        button.addEventListener('click', () => {
          state.genreKey = key;
          resetTest(true);
          renderGenres();
        });
        els.genreWrap.appendChild(button);
      });
    }

    function renderPassage() {
      els.typingText.innerHTML = '';
      for (let i = 0; i < state.passage.length; i++) {
        const span = document.createElement('span');
        span.className = 'char pending';
        const typedChar = state.typed[i];
        if (typedChar != null) {
          span.className = typedChar === state.passage[i] ? 'char correct' : 'char incorrect';
        }
        span.textContent = state.passage[i];
        els.typingText.appendChild(span);
      }
    }

    function updateStatsUI() {
      const stats = calculateStats();
      const timeText = `${state.elapsed.toFixed(1)}s`;

      els.topWpm.textContent = stats.wpm;
      els.topAcc.textContent = `${stats.accuracy}%`;
      els.topTime.textContent = timeText;

      els.metricWpm.textContent = stats.wpm;
      els.metricAcc.textContent = `${stats.accuracy}%`;
      els.metricTime.textContent = timeText;

      els.progressText.textContent = `Progress ${stats.progress}%`;

      if (state.finished) {
        els.resultBox.classList.add('show');
        els.resultWpm.textContent = stats.wpm;
        els.resultAcc.textContent = `${stats.accuracy}%`;
        els.resultTime.textContent = timeText;
      } else {
        els.resultBox.classList.remove('show');
      }
    }

    function updateStatus() {
      const country = selectedCountry();
      els.languageText.textContent = `${country.flag} ${country.language} • safe optimistic content`;
    }

    function stopTimer() {
      if (state.timer) {
        clearInterval(state.timer);
        state.timer = null;
      }
    }

    function startTimer() {
      if (state.timer || state.finished) return;
      state.startedAt = Date.now();
      state.timer = setInterval(() => {
        state.elapsed = (Date.now() - state.startedAt) / 1000;
        updateStatsUI();
      }, 100);
    }

    function resetTest(makeNewPassage = false) {
      stopTimer();
      state.typed = '';
      state.startedAt = null;
      state.elapsed = 0;
      state.finished = false;
      if (makeNewPassage || !state.passage) {
        state.passage = buildPassage(state.genreKey);
      }
      els.typingInput.value = '';
      renderPassage();
      updateStatsUI();
      updateStatus();
      requestAnimationFrame(() => els.typingInput.focus());
    }

    function handleInput(event) {
      if (state.finished) return;
      const value = event.target.value;
      if (!state.startedAt && value.length > 0) startTimer();
      state.typed = value;
      renderPassage();
      if (state.typed === state.passage && state.passage.length > 0) {
        state.finished = true;
        stopTimer();
      }
      updateStatsUI();
    }

    els.generateBtn.addEventListener('click', () => resetTest(true));
    els.typingInput.addEventListener('input', handleInput);
    els.typingText.addEventListener('click', () => els.typingInput.focus());

    renderFlags();
    renderSelectedCountry();
    renderGenres();
    resetTest(true);
  </script>
</body>
</html>
